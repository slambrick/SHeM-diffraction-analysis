A module for importing, analysing, and plotting SHeM spot profile data.

The 2D Gaussian function is based on this StackOverflow post.

A model for the Parallel speed ratio is used from M Bergin 2017. and the data that was used to fit to Bergin's model is from Toennies & Winkelman.

SHeM (design for Cambridge A-SHeM): http://doi.org/10.1016/j.nimb.2014.06.028

2D diffraction with SHeM: http://doi.org/coming_soon!

  51def energy_from_T(T):
  52    """Calculates incident helium energy in J from the beam temperature  - 
  53    assumes a pure beam.
  54
  55    Parameters
  56    ----------
  57    T : float, int, numpy.ndarray
  58        The beam temperature in K.
  59
  60    Returns
  61    -------
  62    E: float, numpy.ndarray
  63        The average energy of helium atoms in the beam, in J.
  64    """
  65    E = 5*k_B*T/2 # J
  66    return(E)

  69def effective_T(E):
  70    """Calculates the 'effective temperature' for the beam, that is the
  71    equivalent temperature for a pure helium beam with this energy. Accepts
  72    energy in meV.
  73
  74    Parameters
  75    ----------
  76    E : float, int, numpy.ndarray
  77        The average energy of helium atoms in a beam, in J.
  78
  79    Returns
  80    -------
  81    T: float, numpy.ndarray
  82        The temperature a pure helium beam would have to be at to give the
  83        atoms in the beam an energy of `E`. In units of K.
  84    """
  85    E = E*1.6e-19/1000

  90def speed_ratio(P_0, d_noz):
  91    """Calculates the parallel speed ratio for the specified nozzle
  92    pressure and diameter for a helium beam based on an emprical fit.
  93    
  94    The fit is from [M.Bergin 2018](http://doi.org/10.1016/j.ultramic.2019.112833)
  95    and is to data/simulation from [Toennies & Winkelman](http://doi.org/10.1063/1.434448).
  96    
  97
  98    Parameters
  99    ----------
 100    P_0 : float, int, numpy.ndarray
 101        Pressure of the beam, specified in torr.
 102    d_noz : float, int, numpy.ndarray
 103        Diameter of the nozzle, specified in cm.
 104
 105    Returns
 106    -------
 107    sR : float, numpy.ndarray
 108        The predicted terminal, parallel speed ratio.
 109    """
 110    a = 0.43
 111    b = 0.76
 112    c = 0.84
 113    d = 5.2
 114    mu = 1.97
 115    lS = a*np.log10(P_0*d_noz) + b + c/(1 + np.exp(-d*(np.log10(P_0*d_noz) - mu)))
 116    sR = 10**lS
 117    return(sR)

 120def morse_V(z, x, y, params=(8.03, 1.35, 1.08, 0.102, 4)):
 121    """Calculates a corrugated Morse potential, from eq. 1.2 & 1.3 [Celli et al.
 122    1995](https://doi.org/10.1063/1.449297).
 123    
 124    $$ Q(x,y) = h \\left( \cos \dfrac{2\pi x}{a} + \cos \dfrac{2\pi y}{a} \\right)$$
 125    $$ V(x, y, z) = D_0 e^{-2 \\alpha [z - Q(x, y)]} - 2 D_0 e^{-\\alpha_1 z} $$
 126
 127    Parameters
 128    ----------
 129    z : float, int, numpy.ndarray
 130        The height above the surface, in nm.
 131    x : float, int, numpy.ndarray
 132        x position on the surface, in nm.
 133    y : float, int, numpy.ndarray
 134        y position on the surface, in nm.
 135    params : 5 tupe of float/int, optional
 136        Parameters for the Morse potential specified as a 5 element tuple,
 137        (D<sub>0</sub>, , <sub>1</sub>, h, a). The default is (8.03, 1.35, 1.08, 0.102, 4).
 138
 139    Returns
 140    -------
 141    V: float, numpy.ndarray
 142        Value of the potential.
 143    """
 144    D0, alpha, alpha1, h, a = params
 145    Q = h*(np.cos(2*pi*x/a) + np.cos(2*pi*y/a))
 146    v = D0*(np.exp(-2*alpha*(z - Q)) - 2*np.exp(-alpha1*z))
 147    return(v)

 150def twoD_Gaussian(xy, amplitude, xo, yo, sigma_x, sigma_y, theta, f, g, h, k, l, m):
 151    """A two dimensional gaussian function for fitting to diffraction peaks.
 152    The background is defined in the `background` function.
 153
 154    $$ a = \dfrac{\cos^2 \\theta}{2 \sigma_x^2} + \dfrac{\sin^2 \\theta}{2\sigma_y^2}$$
 155    $$ b = -\dfrac{\sin 2 \\theta}{4\sigma_x^2} + \dfrac{\sin 2 \\theta}{4\sigma_y^2}$$
 156    $$ c =  \dfrac{\sin^2 \\theta}{2 \sigma_x^2} + \dfrac{\cos^2 \\theta}{2\sigma_y^2}$$
 157    $$ I(x, y) = B + A e^{-a(x - x_0)^2 - 2b(x-x_0)(y-y_0) - c(y - y_0)^2}$$
 158    
 159    Parameters
 160    ----------
 161    xy : 2 tuple of float, int, numpy.ndarray
 162        tuple of x & y coordinates.
 163    amplitude : float, int
 164        height of the Gaussian.
 165    xo : float, int
 166        x coordinate of the centre.
 167    yo : float, int
 168        y coordinate of the centre.
 169    sigma_x : float, int
 170        x direction standard deviation.
 171    sigma_y : float, int
 172        y direction standard deviation.
 173    theta : float, int
 174        clockwise rotation (rad) of the Gaussian.
 175    f : float, int
 176        1st Coefficient for the background.
 177    g : float, int
 178        2nd Coefficient for the background.
 179    h : float, int
 180        3rd Coefficient for the background.
 181    k : float, int
 182        4th Coefficient for the background.
 183    l : float, int
 184        5th Coefficient for the background.
 185    m : float, int
 186        6th Coefficient for the background.
 187
 188    Returns
 189    -------
 190    popt : scipy fitting output
 191        DESCRIPTION
 192    pconv : scipy fitting output
 193        DESCRIPTION.
 194
 195    """
 196    
 197    x, y = xy
 198    xo = float(xo)
 199    yo = float(yo)    
 200    a = (np.cos(theta)**2)/(2*sigma_x**2) + (np.sin(theta)**2)/(2*sigma_y**2)
 201    b = -(np.sin(2*theta))/(4*sigma_x**2) + (np.sin(2*theta))/(4*sigma_y**2)
 202    c = (np.sin(theta)**2)/(2*sigma_x**2) + (np.cos(theta)**2)/(2*sigma_y**2)
 203    tot = amplitude*np.exp( - (a*((x-xo)**2) + 2*b*(x-xo)*(y-yo) 
 204                            + c*((y-yo)**2)))
 205    tot = tot + background(xy, f, g, h, k, l, m)# f + g*x + h*y + k*x*y + l*x**2 + m*y**2
 206    return(tot.ravel())

 209def background(xy, f, g, h, k, l, m):
 210    """2nd order 2D polynomial background function.
 211    
 212    $$ b(x,y) = f + gx + hy + kxy + lx^2 + my^2 $$
 213    
 214
 215    Parameters
 216    ----------
 217    xy : tuple
 218        Tuple of `(x,y)` values, either single numerical values or 
 219        numpy.ndarray.
 220    f : float, int
 221        1st Coefficient for the background..
 222    g : float, int
 223        2nd Coefficient for the background..
 224    h : float, int
 225        3rd Coefficient for the background..
 226    k : float, int
 227        4th Coefficient for the background..
 228    l : float, int
 229        5th Coefficient for the background..
 230    m : float, int
 231        6th Coefficient for the background..
 232
 233    Returns
 234    -------
 235    b : float, numpy.ndarray
 236        Value of the 2nd order polynomial.
 237    """
 238    
 239    x, y = xy
 240    b = f + g*x + h*y + k*x*y + l*x**2 + m*y**2
 241    return(b)

 244def theta_of_z(z, plate="C06", WD=2):
 245    '''Calculates the detection angle theta from the z position in mm
 246    for the 1st gneration angular resolution pinhole plate for the Cambridge 
 247    A-SHeM. plate specification: #C06
 248    
 249    Can also calculate the detection angle from the z position in mm for the 
 250    mixed gas pinhole plate for the Cambridge B-SHeM. plate specification: #Z07
 251    
 252    Alternaticly: calculates the detection angle, theta, from the z position in mm for a
 253    45deg incidnece and 45deg detection pinhole plate with the specified design
 254    working distance WD
 255    '''
 256    if plate == "C06":
 257        return(np.arctan((7 - (z+1.5))/(z+1.5))*180/pi)
 258    elif plate == "standard":
 259        return(np.arctan((2*WD - z)/z)*180/pi)
 260    elif plate == "Z07":
 261        L = 6*np.tan(30*pi/180)
 262        return(np.arctan((L - z*np.tan(30*pi/180))/z)*180/pi) #TOOD
 263    else:
 264        raise ValueError('Unkwon pinhole plate, known: "C06", "Z07", "standard')

 267def load_z_scans(files, z_zero = 3.41e6, instrument = 'ashem'):
 268    """Loads a series of z scans taken at different aximulath orientations. The
 269    orientations need to provied seperatly as they are not stored in the data
 270    files. This file is written specifically for the Cambridge A-SHeM data 
 271    taken between 2020-23.
 272    
 273
 274    Parameters
 275    ----------
 276    files : array_like of str
 277        The files that store the z scan data.
 278    z_zero : float, int, optional
 279        The "z zero": the value of z in stage coordinates that corresponds to 
 280        a distance of 0 from the pinhole plate. The default is 3.41e6.
 281    instrument : str, optional
 282        Which instrument? "ashem" or "bshem". The default is 'ashem'.
 283
 284    Returns
 285    -------
 286    zs: numpy.ndarray
 287        The z values, in mm, for these z scans.
 288    meas: dict
 289    
 290    example_pos: numpy.ndarray 
 291
 292    Note the contents of a Z-scan file for A-SHeM:  
 293        `z_positions`, `inputs`, `detector_mode`, `N_dwell`, `sampling_period`, `date_time`
 294        `det_params`, `p1`, `p2`, `p3`, `p_srce`, `p_diff`, `p_smpl`, `p_det`, `pause`
 295        `rotation_angle`, `counts`, `errors`, `finish_time`, `scan_time`
 296    """
 297   
 298    # Import the meas structure
 299    meas = scipy.io.loadmat(files[0])['meas']
 300    
 301    # Get the z positions (the same for all the data files
 302    # A-SHeM has positive z towards the pinhole plate
 303    # B-SHeM has positive z away from the pinhole plate
 304    if instrument == 'ashem':
 305        zs = (z_zero - meas['z_positions'][0][0]) 
 306    elif instrument == 'bshem': 
 307        zs = (meas['z_positions'][0][0] - z_zero) 
 308    else: 
 309        raise('Unkown instrument input in load_z_scans') 
 310    for j, item in enumerate(meas['inputs'][0][0][0]):
 311        if isinstance(item[0], str):
 312            if item[0] == 'example_pos':
 313                example_pos = meas['inputs'][0][0][0][j+1][0]
 314                break
 315
 316    return(zs, meas, example_pos)

 376def add_scale(ax, label, x_offset=0.08):
 377    '''A function to add an extra x scale to the polar plot with the provided
 378    axis handle. The offset determines where the scale is drawn.
 379    
 380    Inputs:
 381        ax - axis handle on which to add an extra scale
 382        label - text label for the new scale
 383        x_offset - how far to the left of the plot to add the new scale
 384    '''
 385    
 386    # add extra axes for the scale
 387    rect = ax.get_position()
 388    rect = (rect.xmin-x_offset, rect.ymin+rect.height/2, # x, y
 389            rect.width, rect.height/2) # width, height
 390    scale_ax = ax.figure.add_axes(rect)
 391    # hide most elements of the new axes
 392    for loc in ['right', 'top', 'bottom']:
 393        scale_ax.spines[loc].set_visible(False)
 394    scale_ax.tick_params(bottom=False, labelbottom=False)
 395    scale_ax.patch.set_visible(False) # hide white background
 396    # adjust the scale
 397    scale_ax.spines['left'].set_bounds(*ax.get_ylim())
 398    scale_ax.set_yticks(ax.get_yticks())
 399    scale_ax.set_ylim(ax.get_rorigin(), ax.get_rmax())
 400    scale_ax.set_ylabel(label)
 401    return(scale_ax)

 403def find_diff_peak(data, kx, ky, plotit=False, i=np.nan):
 404    """Try to fit a 2D gaussian to the diffraction peak"""
 405    
 406    # The grid to be used for fitting, we isolate a single peak
 407    # to make the fitting easier
 408    I, kxx, kyy = data.grid_interpolate(kx = (kx-8, kx+8), 
 409                                            ky = (ky-8, ky+8), 
 410                                            N=101, method='linear')
 411    if plotit:
 412        f, axs = plt.subplots(2, 2, figsize=(10, 10))
 413        ax1, ax2 = axs[0]
 414        ax3, ax4 = axs[1]
 415        # Identify a single diffraction peak
 416        data.interpolated_plot(kx = (kx-8, kx+8), ky = (ky-8, ky+8), N = 101, method = 'nearest', ax=axs[0,0])
 417    # Initial guess for the height is the range
 418    height = np.max(I) - np.min(I)
 419    # Initial guess for the offset is the minimum
 420    offset = np.min(I)
 421    # Initial guess for the standard deviations is 3x3
 422    # Initial guess for the orinentation is 45deg (this is the most
 423    # dodgy one)
 424    initial_guess = (height,kx,ky,3,3,-45*np.pi/180,offset, 0, 0, 0, 0, 0)
 425    try:
 426        popt, pcov = curve_fit(twoD_Gaussian, (kxx, kyy), 
 427                                   I.flatten(), p0=initial_guess)
 428    except:
 429        popt = np.ones(12)*np.nan
 430    
 431    if plotit:
 432        # Add the identified centre to the plot of the raw data
 433        axs[0,0].plot(popt[1], popt[2], 'ro')
 434
 435        # Plot the fitted Gaussian (does it look decent at all?)
 436        Gau = twoD_Gaussian((kxx, kyy), popt[0], popt[1], popt[2], popt[3], popt[4], popt[5], 0, 0, 0, 0, 0, 0)
 437        axs[1,0].pcolormesh(kxx, kyy, Gau.reshape(101, 101), edgecolors='face')
 438        axs[1,0].axis('equal')
 439        axs[1,0].set_xlabel('$k_x/\mathrm{nm}^{-1}$')
 440        axs[1,0].set_ylabel('$k_y/\mathrm{nm}^{-1}$')
 441        axs[1,0].set_title('Fitted Gaussian')
 442
 443        # Plot the background (does it look decent at all?)
 444        Back = background((kxx, kyy), popt[6], popt[7], popt[8], popt[9], popt[10], popt[11])
 445        axs[1,1].pcolormesh(kxx, kyy, Back.reshape(101, 101), edgecolors='face')
 446        axs[1,1].axis('equal')
 447        axs[1,1].set_xlabel('$k_x/\mathrm{nm}^{-1}$')
 448        axs[1,1].set_ylabel('$k_y/\mathrm{nm}^{-1}$')
 449        axs[1,1].set_title('Fitted Polynomial Background')
 450
 451        # Plot the background (does it look decent at all?)
 452        GauTot = twoD_Gaussian((kxx, kyy), *popt)
 453        axs[0,1].pcolormesh(kxx, kyy, GauTot.reshape(101, 101), edgecolors='face')
 454        axs[0,1].axis('equal')
 455        axs[0,1].set_xlabel('$k_x/\mathrm{nm}^{-1}$')
 456        axs[0,1].set_ylabel('$k_y/\mathrm{nm}^{-1}$')
 457        axs[0,1].set_title('Total fit')
 458        plt.savefig('all_fits/peak_fitted{:03d}.png'.format(i))
 459    return(popt)

 809    def calc_dK(self):
 810        """Calculates the in plane momentum transfer for the data file and the
 811        projected in plane momentum transfer (psuedo) k<sub>x</sub>, 
 812        k<sub>y</sub>. Values are calculated in nm<sup>-1</sup>.
 813        
 814
 815        Returns
 816        -------
 817        None.
 818
 819        """
 820        K = 2*pi*sqrt(2*m_He*self.E)/h # m^-1
 821        #K = k*np.sin(incident_angle*pi/180) 
 822        #K = 2*pi*sqrt(5*m_He*k_B*self.T)/h; # m^-1
 823        # Calculates the parallel momentum transfer in nm^-1
 824        self.DK = K*(np.sin(self.theta*pi/180) - np.sin(self.incident_angle*pi/180) )/1e9;
 825        # Calculate the projected k values
 826        self.kx = -K*( (np.sin(self.theta*pi/180) - np.sin(self.incident_angle*pi/180) )*np.cos(self.alpha*pi/180) )/1e9;
 827        self.ky = -K*(np.sin(self.theta*pi/180) - np.sin(self.incident_angle*pi/180) )*np.sin(self.alpha*pi/180)/1e9;
