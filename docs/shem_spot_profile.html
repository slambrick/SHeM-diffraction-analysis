 48def energy_from_T(T):
 49    """Calculates incident helium energy in J from the beam temperature  - 
 50    assumes a pure beam.
 51
 52    Parameters
 53    ----------
 54    T : float, int
 55        The beam temperature in K.
 56
 57    Returns
 58    -------
 59    E: float
 60        The average energy of helium atoms in the beam, in J.
 61    """
 62    E = 5*k_B*T/2 # J
 63    return(E)

 66def effective_T(E):
 67    """Calculates the 'effective temperature' for the beam, that is the
 68    equivalent temperature for a pure helium beam with this energy. Accepts
 69    energy in meV.
 70
 71    Parameters
 72    ----------
 73    E : float, int
 74        The average energy of helium atoms in a beam, in J.
 75
 76    Returns
 77    -------
 78    T: float
 79        The temperature a pure helium beam would have to be at to give the
 80        atoms in the beam an energy of `E`. In units of K.
 81    """
 82    E = E*1.6e-19/1000
 83    T = 2*E/(5*k_B)
 84    return(T)

 87def speed_ratio(P_0, d_noz):
 88    """Calculates the parallel speed ratio for the specified nozzle
 89    pressure and diameter for a helium beam based on an emprical fit.
 90    
 91    The fit is from [M.Bergin 2018](http://doi.org/10.1016/j.ultramic.2019.112833)
 92    and is to data/simulation from [Toennies & Winkelman](http://doi.org/10.1063/1.434448).
 93    
 94
 95    Parameters
 96    ----------
 97    P_0 : float, int
 98        Pressure of the beam, specified in torr.
 99    d_noz : float, int
100        Diameter of the nozzle, specified in cm.
101
102    Returns
103    -------
104    sR : float
105        The predicted terminal, parallel speed ratio.
106    """
107    a = 0.43
108    b = 0.76
109    c = 0.84
110    d = 5.2
111    mu = 1.97
112    lS = a*np.log10(P_0*d_noz) + b + c/(1 + np.exp(-d*(np.log10(P_0*d_noz) - mu)))
113    sR = 10**lS
114    return(sR)

117def morse_V(z, x, y, params=(8.03, 1.35, 1.08, 0.102, 4)):
118    """Calculates a corrugated Morse potential, from eq. 1.2 & 1.3 Celli et al.
119    1995 https://doi.org/10.1063/1.449297.
120    
121    Inputs:
122        z - height from surface
123        x - lateral x position
124        y - lateral y position
125        params - 5 tuple, potential parameters (D0, alpha, alpha1, h, a)
126    
127    """
128    D0, alpha, alpha1, h, a = params
129    Q = h*(np.cos(2*pi*x/a) + np.cos(2*pi*y/a))
130    v = D0*(np.exp(-2*alpha*(z - Q)) - 2*np.exp(-alpha1*z))
131    return(v)

134def twoD_Gaussian(xy, amplitude, xo, yo, sigma_x, sigma_y, theta, f, g, h, k, l, m):
135    '''A two dimensional gaussian function for fitting to diffraction peaks.
136    
137    Inputs:
138        xy - tuple of x & y coordinates
139        amplitude - height of the Gaussian
140        x0 - x coordinate of the centre
141        y0 - y coordinate of the centre
142        sigma_x - x direction standard deviation
143        sigma_y - y direction standard deviation
144        theta - clockwise rotation (rad) of the Gaussian
145        f,g,h,k,l,m - parameters for a 2nd order polynomial background
146    '''
147    
148    x, y = xy
149    xo = float(xo)
150    yo = float(yo)    
151    a = (np.cos(theta)**2)/(2*sigma_x**2) + (np.sin(theta)**2)/(2*sigma_y**2)
152    b = -(np.sin(2*theta))/(4*sigma_x**2) + (np.sin(2*theta))/(4*sigma_y**2)
153    c = (np.sin(theta)**2)/(2*sigma_x**2) + (np.cos(theta)**2)/(2*sigma_y**2)
154    tot = amplitude*np.exp( - (a*((x-xo)**2) + 2*b*(x-xo)*(y-yo) 
155                            + c*((y-yo)**2)))
156    tot = tot + f + g*x + h*y + k*x*y + l*x**2 + m*y**2
157    return tot.ravel()

166def theta_of_z(z, plate="C06", WD=2):
167    '''Calculates the detection angle theta from the z position in mm
168    for the 1st gneration angular resolution pinhole plate for the Cambridge 
169    A-SHeM. plate specification: #C06
170    
171    Can also calculate the detection angle from the z position in mm for the 
172    mixed gas pinhole plate for the Cambridge B-SHeM. plate specification: #Z07
173    
174    Alternaticly: calculates the detection angle, theta, from the z position in mm for a
175    45deg incidnece and 45deg detection pinhole plate with the specified design
176    working distance WD
177    '''
178    if plate == "C06":
179        return(np.arctan((7 - (z+1.5))/(z+1.5))*180/pi)
180    elif plate == "standard":
181        return(np.arctan((2*WD - z)/z)*180/pi)
182    elif plate == "Z07":
183        L = 6*np.tan(30*pi/180)
184        return(np.arctan((L - z*np.tan(30*pi/180))/z)*180/pi) #TOOD
185    else:
186        raise ValueError('Unkwon pinhole plate, known: "C06", "Z07", "standard')

189def load_z_scans(files, z_zero = 3.41e6, instrument = 'ashem'):
190    """Loads a series of z scans taken at different aximulath orientations. The
191    orientations need to provied seperatly as they are not stored in the data
192    files. This file is written specifically for the Cambridge A-SHeM data 
193    taken between 2020-23.
194    
195
196    Parameters
197    ----------
198    files : array_like of str
199        The files that store the z scan data.
200    z_zero : float, int, optional
201        The "z zero": the value of z in stage coordinates that corresponds to 
202        a distance of 0 from the pinhole plate. The default is 3.41e6.
203    instrument : str, optional
204        Which instrument? "ashem" or "bshem". The default is 'ashem'.
205
206    Returns
207    -------
208    zs: numpy.ndarray
209        The z values, in mm, for these z scans.
210    meas: dict
211    
212    example_pos: numpy.ndarray 
213
214    Note the contents of a Z-scan file for A-SHeM:  
215        `z_positions`, `inputs`, `detector_mode`, `N_dwell`, `sampling_period`, `date_time`
216        `det_params`, `p1`, `p2`, `p3`, `p_srce`, `p_diff`, `p_smpl`, `p_det`, `pause`
217        `rotation_angle`, `counts`, `errors`, `finish_time`, `scan_time`
218    """
219   
220    # Import the meas structure
221    meas = scipy.io.loadmat(files[0])['meas']
222    
223    # Get the z positions (the same for all the data files
224    # A-SHeM has positive z towards the pinhole plate
225    # B-SHeM has positive z away from the pinhole plate
226    if instrument == 'ashem':
227        zs = (z_zero - meas['z_positions'][0][0]) 
228    elif instrument == 'bshem': 
229        zs = (meas['z_positions'][0][0] - z_zero) 
230    else: 
231        raise('Unkown instrument input in load_z_scans') 
232    for j, item in enumerate(meas['inputs'][0][0][0]):
233        if isinstance(item[0], str):
234            if item[0] == 'example_pos':
235                example_pos = meas['inputs'][0][0][0][j+1][0]
236                break
237
238    return(zs, meas, example_pos)

241def load_z_scans_ashem(files_ind, path_name, z_zero = 3.41e6):
242    ashem_fname = np.vectorize(lambda ind, p : '{}/Z{}.mat'.format(p, str(ind).zfill(6)))
243    files = ashem_fname(files_ind, path_name)
244    zs, meas, example_pos = load_z_scans(files, z_zero = z_zero, instrument = 'ashem')
245    # Load in the data
246    # Note the [0][0] needed to extract the data from the Matlab scruct
247    data = np.zeros([len(files_ind), len(meas['counts'][0][0])])
248    for i, f in enumerate(files):
249        meas = scipy.io.loadmat(f)['meas']

251    
252    # A-SHeM uses nm
253    return({'zs' : zs*1e-6, 'I' : data.transpose(), 'example_pos' : example_pos})

298def add_scale(ax, label, x_offset=0.08):
299    '''A function to add an extra x scale to the polar plot with the provided
300    axis handle. The offset determines where the scale is drawn.

305        x_offset - how far to the left of the plot to add the new scale
306    '''
307    
308    # add extra axes for the scale
309    rect = ax.get_position()
310    rect = (rect.xmin-x_offset, rect.ymin+rect.height/2, # x, y
311            rect.width, rect.height/2) # width, height
312    scale_ax = ax.figure.add_axes(rect)

319    scale_ax.spines['left'].set_bounds(*ax.get_ylim())
320    scale_ax.set_yticks(ax.get_yticks())
321    scale_ax.set_ylim(ax.get_rorigin(), ax.get_rmax())
322    scale_ax.set_ylabel(label)
